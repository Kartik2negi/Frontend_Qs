## 📄 File 2: src/controllers/chat.controller.js

```javascript
import openai from '../services/openai.service.js';

/**
 * POST /api/chat
 * Basic chat endpoint
 * 
 * Body: { "prompt": "Explain React" }
 * 
 * Response: {
 *   success: true,
 *   answer: "...",
 *   tokens: { input, output, total }
 * }
 */
export const chat = async (req, res) => {
  try {
    const { prompt } = req.body;

    // Validation
    if (!prompt || typeof prompt !== 'string') {
      return res.status(400).json({
        success: false,
        message: 'Invalid request. "prompt" is required and must be a string.'
      });
    }

    if (prompt.length > 4000) {
      return res.status(400).json({
        success: false,
        message: 'Prompt too long. Maximum 4000 characters.'
      });
    }

    // Call OpenAI
    const response = await openai.chat.completions.create({
      model: 'gpt-3.5-turbo',
      messages: [
        {
          role: 'user',
          content: prompt
        }
      ],
      temperature: 0.7,
      max_tokens: 1000
    });

    // Extract answer
    const answer = response.choices[0].message.content;

    // Log
    console.log(`✓ Chat: ${response.usage.total_tokens} tokens used`);

    // Return
    res.json({
      success: true,
      prompt,
      answer,
      tokens: {
        input: response.usage.input_tokens,
        output: response.usage.output_tokens,
        total: response.usage.total_tokens
      },
      model: 'gpt-3.5-turbo'
    });

  } catch (error) {
    console.error('❌ Error:', error.message);

    if (error.status === 401) {
      return res.status(401).json({
        success: false,
        message: 'Invalid API key'
      });
    }

    if (error.status === 429) {
      return res.status(429).json({
        success: false,
        message: 'Rate limited. Too many requests.'
      });
    }

    res.status(500).json({
      success: false,
      message: 'Something went wrong',
      error: error.message
    });
  }
};

/**
 * POST /api/frontend-interviewer
 * Acts as a senior frontend engineer conducting interview
 * 
 * Body: { "question": "Explain React hooks" }
 * 
 * Response: {
 *   interviewer: "Frontend Engineer",
 *   response: "..."
 * }
 */
export const frontendInterviewer = async (req, res) => {
  try {
    const { question } = req.body;

    if (!question) {
      return res.status(400).json({
        success: false,
        message: 'Question is required'
      });
    }

    const response = await openai.chat.completions.create({
      model: 'gpt-4o',
      messages: [
        {
          role: 'system',
          content: `You are a Senior Frontend Engineer with 12 years of experience.
                   You are conducting a technical interview for a mid-level developer position.
                   
                   Your interview style:
                   - Professional but approachable
                   - Explain why certain approaches are better
                   - Ask follow-up questions to gauge depth
                   - Mention real-world scenarios
                   - Be encouraging even when pointing out gaps
                   
                   Your areas of expertise:
                   - React (hooks, performance, state management)
                   - TypeScript
                   - Web standards (HTML, CSS, JavaScript)
                   - Performance optimization
                   - Testing`
        },
        {
          role: 'user',
          content: question
        }
      ],
      temperature: 0.8,
      max_tokens: 1500
    });

    console.log(`✓ Interview: ${response.usage.total_tokens} tokens`);

    res.json({
      success: true,
      role: 'Frontend Engineer',
      question,
      response: response.choices[0].message.content,
      tokens: response.usage.total_tokens
    });

  } catch (error) {
    console.error('Interview error:', error.message);
    res.status(500).json({
      success: false,
      message: 'Error',
      error: error.message
    });
  }
};

/**
 * POST /api/resume-reviewer
 * Acts as an experienced resume reviewer
 * 
 * Body: { "resume": "Senior Engineer..." }
 * 
 * Response: {
 *   feedback: "..."
 * }
 */
export const resumeReviewer = async (req, res) => {
  try {
    const { resume } = req.body;

    if (!resume || resume.length < 50) {
      return res.status(400).json({
        success: false,
        message: 'Resume must be at least 50 characters'
      });
    }

    if (resume.length > 10000) {
      return res.status(400).json({
        success: false,
        message: 'Resume too long (max 10,000 characters)'
      });
    }

    const response = await openai.chat.completions.create({
      model: 'gpt-4o',
      messages: [
        {
          role: 'system',
          content: `You are an expert Resume Reviewer with 15 years of HR experience.
                   
                   When reviewing a resume, provide feedback on:
                   1. Format and organization (is it scannable?)
                   2. Content quality (are achievements quantified?)
                   3. ATS optimization (will it pass filters?)
                   4. Clarity and professionalism
                   5. Specific, actionable suggestions
                   
                   Be constructive, encouraging, and specific.
                   Start with strengths, then suggest improvements.
                   End with top 3 actionable items to improve.`
        },
        {
          role: 'user',
          content: `Please review this resume:\n\n${resume}`
        }
      ],
      temperature: 0.7,
      max_tokens: 1500
    });

    console.log(`✓ Resume review: ${response.usage.total_tokens} tokens`);

    res.json({
      success: true,
      feedback: response.choices[0].message.content,
      tokens: response.usage.total_tokens
    });

  } catch (error) {
    console.error('Resume review error:', error.message);
    res.status(500).json({
      success: false,
      message: 'Error',
      error: error.message
    });
  }
};
```

## 📄 File 4: src/app.js

```javascript
import express from 'express';
import cors from 'cors';
import dotenv from 'dotenv';
import chatRoutes from './routes/chat.route.js';

// Load environment variables
dotenv.config();

const app = express();
const PORT = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(express.json());

// Health check
app.get('/health', (req, res) => {
  res.json({ status: '✓ Server is running' });
});

// Home
app.get('/', (req, res) => {
  res.json({
    message: 'AI Chat Backend',
    endpoints: {
      health: 'GET /health',
      chat: 'POST /api/chat',
      interviewer: 'POST /api/frontend-interviewer',
      reviewer: 'POST /api/resume-reviewer'
    }
  });
});

// Routes
app.use('/api', chatRoutes);

// 404
app.use((req, res) => {
  res.status(404).json({
    success: false,
    message: 'Endpoint not found'
  });
});

// Error handler
app.use((error, req, res, next) => {
  console.error('Error:', error);
  res.status(500).json({
    success: false,
    message: 'Server error',
    error: process.env.NODE_ENV === 'development' ? error.message : undefined
  });
});

// Start
app.listen(PORT, () => {
  console.log(`
  ╔════════════════════════════════════╗
  ║  🚀 AI Backend Started             ║
  ╚════════════════════════════════════╝
  
  📍 http://localhost:${PORT}
  
  📝 Endpoints:
    • POST /api/chat
    • POST /api/frontend-interviewer
    • POST /api/resume-reviewer
  
  🔍 Health: GET /health
  `);
});
```

## 🔧 Troubleshooting

### Error: "Cannot find module 'openai'"

```bash
npm install openai
```

### Error: "API key not found"

Check your `.env` file:
```env
OPENAI_API_KEY=sk-proj-your-actual-key-here
```

### Error: "401 Unauthorized"

Your API key is invalid. Get a new one from [OpenAI](https://platform.openai.com/api-keys)

### Error: "429 Too Many Requests"

You've hit rate limits. Wait a moment before trying again.
